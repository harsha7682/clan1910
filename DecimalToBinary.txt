#include<stdio.h>
void main(){
    int n,a[10],i=0;
    printf("Enter decimal number : ");
    scanf("%d",&n);
    while(n>0){
        if(n%2==0){
            a[i]=0;
            i++;
            n/=2;
            if(n-2==1){
                a[i]=1;
                a[i+1]=1;
            }
        }
        else{
            a[i]=1;
            i++;
            n/=2;
            if(n-2==1){
                a[i]=1;
                a[i+1]=1;
            }
        }
    }
    i--;
    while(i>=0){
        printf(" %d ",a[i]);
        i--;
    }
}
